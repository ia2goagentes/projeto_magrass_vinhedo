    ```typescript
    // Phase 1 webhook — receives leads from Make (Meta Instant Forms upstream).
    // Contract details in .planning/phases/01-schema-webhook-foundation/01-CONTEXT.md (D-03 through D-07).
    // Runtime MUST be Node.js — timingSafeEqual is not available in Edge runtime.

    export const runtime = "nodejs";

    import { timingSafeEqual } from "crypto";
    import { createServiceClient } from "@/lib/supabase/service";

    function normalizePhone(raw: string): string {
      return raw.replace(/\D/g, "");
    }

    const STANDARD_FIELDS = new Set([
      "name", "nome", "lead_name",
      "phone", "whatsapp", "telefone", "celular",
      "lead_id", "leadId",
    ]);

    export async function POST(request: Request) {
      // D-03: Timing-safe secret comparison. Length must match BEFORE timingSafeEqual
      // (otherwise timingSafeEqual throws TypeError on unequal-length buffers).
      const provided = Buffer.from(request.headers.get("x-webhook-secret") ?? "", "utf8");
      const expected = Buffer.from(process.env.WEBHOOK_SECRET ?? "", "utf8");
      if (
        expected.length === 0 ||
        provided.length !== expected.length ||
        !timingSafeEqual(provided, expected)
      ) {
        return Response.json({ error: "Unauthorized" }, { status: 401 });
      }

      // D-04: 400 for invalid JSON body.
      let body: Record<string, unknown>;
      try {
        body = (await request.json()) as Record<string, unknown>;
      } catch {
        return Response.json({ error: "Invalid JSON" }, { status: 400 });
      }

      // D-05: Normalize field aliases from Make (exact field names TBD — accept common ones).
      const name = (body.name ?? body.nome ?? body.lead_name) as string | undefined;
      const phone = (body.phone ?? body.whatsapp ?? body.telefone ?? body.celular) as
        | string
        | undefined;
      const leadSourceId = (body.lead_id ?? body.leadId) as string | undefined;

      // D-04: 422 for missing required fields (Make does NOT retry on 4xx — do not use 400 here).
      if (!name || !phone) {
        return Response.json(
          { error: "Missing required fields: name, phone" },
          { status: 422 }
        );
      }

      // D-05 / INGST-04: form_answers = everything that isn't a known standard alias.
      const formAnswers: Record<string, unknown> = {};
      for (const [k, v] of Object.entries(body)) {
        if (!STANDARD_FIELDS.has(k)) formAnswers[k] = v;
      }

      const supabase = createServiceClient();

      // D-07 / INGST-02: idempotent upsert. When lead_source_id conflicts,
      // ignoreDuplicates:true → PostgREST ON CONFLICT DO NOTHING → data=[], error=null.
      const { data, error } = await supabase
        .from("leads")
        .upsert(
          {
            lead_source_id: leadSourceId ?? null,
            name,
            whatsapp: normalizePhone(phone), // D-06 / INGST-03
            form_answers: formAnswers,        // INGST-04
            raw_payload: body,
            status: "novo",
            source: "make_webhook",
          },
          { onConflict: "lead_source_id", ignoreDuplicates: true }
        )
        .select("id");

      if (error) {
        // D-04: 500 for DB errors. Never leak Supabase error details in response.
        console.error("Lead insert error", { code: error.code });
        return Response.json({ error: "DB error" }, { status: 500 });
      }

      // ignoreDuplicates:true → empty array when conflict (RESEARCH.md RP-5 + Pitfall 2).
      const isDuplicate = !data || data.length === 0;
      if (isDuplicate) {
        return Response.json({ ok: true, duplicate: true }, { status: 200 });
      }

      return Response.json({ ok: true, id: data[0].id }, { status: 201 });
    }
    ```

    - Use `Response.json(...)` — the Next.js 16 docs example uses the Web `Response` static method.
      Do NOT import NextResponse. `Response.json` is a Web-standard static that returns a Response

    - The `export const runtime = "nodejs"` line must come BEFORE `import` statements only if the

    - Do NOT wrap the handler in try/catch at the top level. Errors from `request.json()` are the
